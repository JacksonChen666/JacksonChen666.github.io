Random Output Volume 
set originalv to output volume of (get volume settings)
set repeatz to the text returned of (display dialog "How many times do you want to set the volume randomly?" default answer "100")
repeat repeatz times
    (set volume output volume random number from 0 to 100)
end repeat
set volume output volume originalv

Random Input Volume 
set originalv to input volume of (get volume settings)
repeat
    try
        set volume input volume (random number from 0 to 100)
        delay (random number from 0.0 to 0.5)
    on error
        set volume input volume originalv
        return originalv
    end try
end repeat

Fun progress bar 
# Save as APP for extra enjoyment
(*
PTS means progress total steps
PD means progress descrition 
PAD means progress additional description
PCS means progress completed steps
PInfo means progress info (additional description)
*)
global PTS, PD, PAD, PCS, numberz, PInfo

on varChecks()
    set PTS to progress total steps
    set PD to progress description
    set PAD to progress additional description
    set PCS to progress completed steps
end varChecks

on setPTS(contentz)
    set progress total steps to contentz
end setPTS

on setPD(contentz)
    set progress description to contentz
end setPD

on setPAD(contentz)
    set progress additional description to contentz
end setPAD

on setPCS(contentz)
    set progress completed steps to contentz
end setPCS

set numberz to {-1, 1}
setPTS(1000)
setPD("Fun Progress Bar")
setPCS(progress total steps / 2)

on flowChecks()
    varChecks()
    if PCS is less than 0 then
        set PCS to setPCS(0)
        set PAD to setPAD("Interger underflow, please wait")
        repeat PTS / 2 times
            set PCS to setPCS(PCS + 1)
            varChecks()
            delay 0.005
        end repeat
        
    else if PCS is greater than PTS - 1 then
        set PCS to setPCS(PTS)
        set PAD to setPAD("Interger overflow, please wait")
        repeat PTS / 2 times
            set PCS to setPCS(PCS - 1)
            varChecks()
            delay 0.005
        end repeat
    end if
    numChecks()
    return
end flowChecks

on numChecks()
    varChecks()
    # Checks the number and adds extra text to it
    if PCS is 500 then
        set PAD to setPAD(PInfo & "(Middle)")
        
    else if PCS is less than 10 then
        set PAD to setPAD(PInfo & "(Extremely Low)")
        
    else if PCS is greater than 990 then
        set PAD to setPAD(PInfo & "(Extremely High)")
        
    else if PCS is less than 50 then
        set PAD to setPAD(PInfo & "(Very Low)")
        
    else if PCS is greater than 950 then
        set PAD to setPAD(PInfo & "(Very High)")
        
    else if PCS is less than 100 then
        set PAD to setPAD(PInfo & "(Low)")
        
    else if PCS is greater than 900 then
        set PAD to setPAD(PInfo & "(High)")
        
    end if
    return
end numChecks

on repeating()
    try
        repeat
            varChecks()
            set PInfo to PCS & "/" & PTS & " steps "
            set PTS to setPCS(PCS + (some item of numberz))
            set PAD to setPAD(PInfo)
            
            flowChecks()
            delay 5.0E-5
        end repeat
    on error the errorMsg number the errorNum
        set PD to setPD("Interrupted by error")
        set PAD to setPAD("Error: " & the errorNum & ". " & the errorMsg)
        if errorNum is -128 then
            return
        else
            display dialog "Error: " & the errorNum & ". " & the errorMsg buttons {"OK"} default button 1 giving up after 10
            return
        end if
        return
    end try
end repeating

repeating()

on quit
    setPTS(0)
    setPD("")
    setPAD("")
    setPCS(0)
    continue quit
end quit

Auto join Zoom 
(* 
It all started when I wanted to join zoom faster than a human could do with no prompt like how I expected like this.
I first discovered zoom.us's URL scheme to join meeting, but when I found out it doesn't work, I started to make this program.
After this all of stuff, I finally have something that is constantly updated.
After this pandemic is over, development might slow down or become discontinued.
*)
#custom breakout room window name is "Zoom - 'breakout room name'"
global meetingNames, meetingIDs, meetingPwds, wname, allowJoin
# Compiling the app will reset all your saved meetings!
property meetingNames : {}
property meetingIDs : {}
property meetingPwds : {}
set allowJoin to true

on joinMeeting(meetingID, meetingPwd)
    tell application "System Events"
        activate application "zoom.us"
        if wname is "Login" then
            try
                my doWithTimeout("click UI element 2 of group 1 of window \"Login\" of application process \"zoom.us\"", 10)
            on error errmsg number errnum
                activate me
                error "Error: " & errnum & ". " & errmsg & "\nUnable to click specified button"
            end try
            try
                click text field 1 of window 1 of application process "zoom.us"
            on error
                my doWithTimeout("click UI element 2 of group 1 of window \"Login\" of application process \"zoom.us\"", 10)
                click text field 1 of window 1 of application process "zoom.us"
            end try
            tell process "zoom.us" to keystroke meetingID
            click UI element 8 of window 1 of application process "zoom.us"
        else if wname is "Zoom" then
            my doWithTimeout("click UI Element 15 of window \"Zoom\" of application process \"zoom.us\"", 10)
            click text field 1 of window 1 of application process "zoom.us"
            tell process "zoom.us" to keystroke meetingID
            click UI element 7 of window 1 of application process "zoom.us"
        else if wname contains "Zoom Meeting" or wname is "Window" then
            activate me
            display dialog "It seems like you are already in a meeting" buttons {"OK"} default button 1
            return false
        else
            error "Please only use while logged out, or logged into zoom. I do not understand the window name, " & wname
            return false
        end if
        
        if meetingPwd is not missing value then # yeah it's all pretty much the same
            try
                my doWithTimeout("click text field 1 of window \"Join a Meeting\" of application process \"zoom.us\"", 10)
            on error errmsg number errnum
                if errmsg is "System Events got an error: Can’t get window \"Join a Meeting\" of application process \"zoom.us\"." then error "Cannot get password window" number errnum
            end try
            tell process "zoom.us" to keystroke meetingPwd
            click UI element "Join" of window "Join a Meeting" of application process "zoom.us"
        end if
    end tell
    return true
end joinMeeting

on lengthCheck()
    if (length of meetingNames) is not equal to (length of meetingIDs) or (length of meetingNames) is not equal to (length of meetingPwds) then
        display dialog "The lists are unequal. Additional items in the list are removed, so please check your list.\nAmounts:\nMeeting Names: " & length of meetingNames & "\nMeeting IDs: " & length of meetingIDs & "\nMeeting passwords: " & length of meetingPwds
        repeat
            if (length of meetingNames) is not equal to (length of meetingIDs) then
                items 1 thru -2 of meetingIDs
            else if (length of meetingNames) is not equal to (length of meetingPwds) then
                items 1 thru -2 of meetingPwds
            else
                exit repeat
            end if
        end repeat
    end if
end lengthCheck

on emergencyMode() # incase it doesn't work
    set extras to {"Add meeting...", "Remove meeting...", "Change meeting info..."}
    activate me
    set choice to choose from list meetingNames & extras default items item 1 of (meetingNames & extras) with prompt "What meeting info would you like to view?"
    if choice is false then # cancel
        return
    else if (choice as text) is "Add meeting..." then
        addToList()
    else if (choice as text) is "Remove meeting..." then
        removeFromList()
    else if (choice as text) is "Change meeting info..." then
        changeInfo()
    else if (choice as text) is "Get Meeting ID and password..." then
        getIDPwd()
    else
        repeat with a from 1 to length of meetingNames
            if (choice as text) is item a of meetingNames then
                showIDPwd(item a of meetingNames, item a of meetingPwds)
                exit repeat
            end if
        end repeat
    end if
    if (choice as text) is "Add meeting..." or (choice as text) is "Remove meeting..." or (choice as text) is "Change meeting info..." or (choice as text) is "Get Meeting ID and password..." then
        userPrompt(versions)
        return
    end if
    return
end emergencyMode

on getWinName() # yes getting the window has been a too hard process
    launch application "zoom.us"
    try
        tell application "System Events"
            tell process "zoom.us" to set wname to name of front window
        end tell
    on error errmsg number errnum
        if errnum is 600 or errmsg is "System Events got an error: Application isn’t running." then # System Events has stopped working
            do shell script "pkill System Events"
            delay 3
            tell application "System Events"
                tell process "zoom.us" to set wname to name of front window
            end tell
        else if errnum is -1719 and errmsg is not "System Events got an error: Can’t get window 1 of process \"zoom.us\". Invalid index." then
            tell application "System Preferences"
                reveal anchor "Privacy" of pane id "com.apple.preference.security"
                display dialog "Please enable accessibility access for me to join zoom for you\nError: " & errmsg & " " & errnum
                activate
            end tell
            set timeOutTime to (current date) + 30
            repeat
                try
                    tell application "System Events"
                        tell process "zoom.us" to set wname to name of front window
                    end tell
                    exit repeat
                on error
                    if ((current date) > timeOutTime) then error errmsg number errnum
                end try
            end repeat
        else if errmsg is "System Events got an error: Can’t get window 1 of process \"zoom.us\". Invalid index." then
            tell application "System Events"
                do shell script "pkill zoom.us"
                launch application "zoom.us"
                set timeOutTime to (current date) + 30
                repeat
                    try
                        tell process "zoom.us" to set wname to name of front window
                        exit repeat
                    on error errmsg number errnum
                        if (current date) > timeOutTime then error errmsg number errnum
                    end try
                end repeat
            end tell
        else
            display dialog "Error: " & errnum & ". " & errmsg & "\nPress OK to continue"
            emergencyMode()
            return
            #error errmsg number errnum
        end if
    end try
    return wname
end getWinName

on userPrompt(versions)
    lengthCheck()
    getWinName()
    if allowJoin is true then
        if wname contains "Zoom Meeting" or wname is "Window" or wname contains "Breakout Room" then
            set allowJoin to false
            activate me
            display dialog "You are already in a meeting, so joining meeting is disabled." buttons {"OK"} default button 1
        else if wname is "" then
            set allowJoin to false
            activate me
            display dialog "I cannot figure which window it is, so I am unable to join for you." buttons {"OK"} default button 1
        else
            set allowJoin to true
        end if
    end if
    set extras to {"Add meeting...", "Remove meeting...", "Change meeting info...", "Get Meeting ID and password...", "I DON'T HAVE TIME TO ADD, JUST JOIN"}
    activate me
    set choice to choose from list meetingNames & extras default items item 1 of (meetingNames & extras)
    if choice is false then # cancel
        return
    else if (choice as text) is "Add meeting..." then
        addToList()
    else if (choice as text) is "Remove meeting..." then
        removeFromList()
    else if (choice as text) is "Change meeting info..." then
        changeInfo()
    else if (choice as text) is "Get Meeting ID and password..." then
        getIDPwd()
    else if (choice as text) is "I DON'T HAVE TIME TO ADD, JUST JOIN" then
        if allowJoin is true then
            noTime()
            return
        else
            display dialog "You can't join meetings because you are already in a meeting" buttons {"OK FINE"} default button 1
            userPrompt(versions)
            return
        end if
    else
        if allowJoin is false then
            display dialog "You are still in the meeting" buttons {"OK"} default button 1
            return
        end if
        repeat with a from 1 to length of meetingNames
            if (choice as text) is item a of meetingNames then
                joinMeeting(item a of meetingIDs, item a of meetingPwds)
                exit repeat
            end if
        end repeat
    end if
    if (choice as text) is "Add meeting..." or (choice as text) is "Remove meeting..." or (choice as text) is "Change meeting info..." or (choice as text) is "Get Meeting ID and password..." or (choice as text) is "I DON'T HAVE TIME TO ADD, JUST JOIN" then
        userPrompt(versions)
        return
    end if
    return
end userPrompt

on getIDPwd()
    if length of meetingNames is 0 then error "There's nothing to see..." number -2573
    set choice to choose from list meetingNames default items item 1 of meetingNames with prompt "What meeting info would you like to view?"
    repeat with a from 1 to length of meetingNames
        if (choice as text) is (item a of meetingNames as text) then
            showIDPwd(item a of meetingNames, item a of meetingIDs, item a of meetingPwds)
            exit repeat
        end if
    end repeat
    return choice
end getIDPwd

on showIDPwd(meetingName, meetingID, meetingPwd)
    if meetingPwd is missing value then
        display dialog "Meeting Name: " & meetingName & "\nMeeting ID: " & meetingID & "\nMeeting link: https://zoom.us/j/" & meetingID buttons {"OK"} default button 1
    else
        display dialog "Meeting Name: " & meetingName & "\nMeeting ID: " & meetingID & "\nMeeting Password: " & meetingPwd & "\nMeeting link: https://zoom.us/j/" & meetingID & "?pwd=" & meetingPwd buttons {"OK"} default button 1
    end if
    return
end showIDPwd

on addToList()
    set meetingNames to meetingNames & repeatUntilAnswered("What meeting name to use?")
    repeat
        set mID to (repeatUntilAnswered("What is the meeting ID?") as string)
        if length of mID is greater than 8 and length of mID is less than 12 then
            exit repeat
        else
            display dialog "Invalid. Meeting IDs must be between 9 to 11 characters long." with icon caution
        end if
    end repeat
    set meetingIDs to meetingIDs & mID
    set mPwd to text returned of (display dialog "What is the meeting password? (Optional, if none, skip, anything that works will work)" default answer "") as string
    if mPwd is "" then set mPwd to missing value
    set meetingPwds to meetingPwds & mPwd
    return
end addToList

on changeInfo()
    if length of meetingNames is 0 then error "There's nothing to see..." number -2573
    set {rename, changeChoice} to {choose from list meetingNames with prompt "Which meeting would you like to change?" default items item 1 of meetingNames, choose from list {"Name", "Meeting ID", "Meeting Password"} with prompt "Which info do you want to change?" default items {"Name"}}
    repeat with a from 1 to length of meetingNames
        if (item a of meetingNames as string) is (rename as string) then
            if (changeChoice as text) is "Name" then
                set item a of meetingNames to text returned of (display dialog "What would you like to rename " & item a of meetingNames & " to?" default answer (item a of meetingNames))
            else if (changeChoice as text) is "Meeting ID" then
                set item a of meetingIDs to text returned of (display dialog "What meeting ID would you like to change to on " & item a of meetingNames & "?" default answer (item a of meetingIDs))
            else if (changeChoice as text) is "Meeting Password" then
                if item a of meetingPwds is missing value then set item a of meetingPwds to ""
                set temp to text returned of (display dialog "What meeting password would you like to change to on " & item a of meetingNames & "?" default answer (item a of meetingPwds))
                if temp is "" then set temp to missing value
                set item a of meetingPwds to temp
            end if
            exit repeat
        end if
    end repeat
    return
end changeInfo

on removeFromList()
    if length of meetingNames is 0 then error "There's nothing to see..." number -2573
    set {toRemove, cleanList, cleanList2, cleanList3} to {choose from list meetingNames with prompt "Remove meeting from list:" default items item 1 of meetingNames, {}, {}, {}}
    repeat with a from 1 to length of meetingNames
        if (item a of meetingNames as string) is not (toRemove as string) then
            set {cleanList, cleanList2, cleanList3} to {cleanList & (item a of meetingNames), cleanList2 & (item a of meetingIDs), cleanList3 & (item a of meetingPwds)}
        end if
    end repeat
    set {meetingNames, meetingIDs, meetingPwds} to {cleanList, cleanList2, cleanList3}
    return {cleanList, cleanList2, cleanList3}
end removeFromList

on versionCheck()
    set {appVersion, supportedVersions, supported, supportedVersion, maxTime} to {version of application "zoom.us", {"5.0"}, false, "", (current date) + 10}
    repeat with a from 1 to length of supportedVersions
        try
            if appVersion contains (item a of supportedVersions as text) then
                set {supported, supportedVersion} to {true, item a of supportedVersions}
                exit repeat
            end if
        on error errmsg number errnum
            if ((current date) > maxTime) then error errmsg number errnum
        end try
    end repeat
    if supported is false then display dialog "Zoom.us version unsupported! You may need to do it manually"
    userPrompt(appVersion)
    return
end versionCheck

on repeatUntilAnswered(prompt)
    repeat
        set temp to text returned of (display dialog prompt default answer "")
        if temp is not "" then exit repeat
    end repeat
    return temp
end repeatUntilAnswered

on doWithTimeout(uiScript, timeoutSeconds)
    set endDate to (current date) + timeoutSeconds
    repeat
        try
            run script "tell application \"System Events\"\n" & uiScript & "\nend tell"
            exit repeat
        on error errorMessage number errnum
            if ((current date) > endDate) then error errorMessage number errnum
        end try
    end repeat
end doWithTimeout

on noTime()
    set ids to text returned of (display dialog "Meeting ID" default answer "")
    if length of ids is greater than 8 and length of ids is less than 12 then
        set pwd to text returned of (display dialog "Meeting password (optional)" default answer "")
        if pwd is "" then set pwd to missing value
        joinMeeting(ids, pwd)
    else
        display dialog "Invalid. Meeting IDs must be between 9 to 11 characters long." with icon caution
        noTime()
    end if
    return
end noTime
